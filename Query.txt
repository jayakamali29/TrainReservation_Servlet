DROP TABLE TRAINERS_TB CASCADE CONSTRAINT;
DROP SEQUENCE TRAINERS_SEQ;

CREATE TABLE TRAINERS_TB(
RECORDID VARCHAR2(12) PRIMARY KEY,
PASSENGERNAME VARCHAR2(40) NOT NULL,
TRAINNUMBER VARCHAR2(20) NOT NULL,
TRAVELDATE DATE NOT NULL,
SEATNO VARCHAR2(10),
TICKETNO VARCHAR2(20),
REMARKS VARCHAR2(100)

);
CREATE SEQUENCE TRAINERS_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99
NOCYCLE;
